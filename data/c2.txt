void vstr_reserve(vstr_t *str, uint32_t resv) {
  if (resv <= str->reserved)
    return;
  resv = resv * 3 / 2;
  assert(resv < VO_STRMAX);
  char *p2 = malloc(resv);
  p2[str->len] = 0;
  if (str->data) {
    memcpy(p2, str->data, str->len);
    free(str->data);
  }
  str->reserved = resv;
  str->data = p2;
}
